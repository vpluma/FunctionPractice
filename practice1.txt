# Global Variables
name = "BILLY"

def function1():
	name = "sammy"
	print(name)

def function2():
	name = "tommy"
	return name

def function3():
	global name
	name = "perry"
	print(name)

print(name)
function1()
print(name)
name = function2()
print(name)
function3()
print(name) #Why didn't this print BILLY?
